import java.util.Scanner;
public class SumAndProduct
{
      public static void main(String[] args)
   {
             Scanner scanner = new Scanner(System.in);
             int n1, n2, sum, product;
             System.out.println("Enter an integer:");
             n1 = scanner.nextInt();
             System.out.println("Enter another integer:");
             n2 = scanner.nextInt();
             sum = n1 + n2;
             product = n1 * n2;
             System.out.println("The sum is "+ sum + ", and the product is "+ product);
             scanner.close();
         
   }
}
import java.util.Scanner;
public class RectangleArea
{
      public static void main(String[] args)
   {
             Scanner scanner = new Scanner(System.in);
             double length, breadth, area;
             System.out.println("Enter length of the rectangle:");
             length = scanner.nextDouble();
             System.out.println("Enter breadth of the rectangle:");
             breadth = scanner.nextDouble();
             area = length * breadth;
             System.out.println("The area of the rectangle is "+ area);
             scanner.close();
         
   }
}
import java.util.Scanner;

public class StudentInfo
{
      public static void main(String[] args)
   {
             Scanner scanner = new Scanner(System.in);
      
             String name;
             int age;
             double gpa;
      
             System.out.println("Please enter student information (Name age GPA) in one line like this: John Smith 20 3.4");
             
             String input = scanner.nextLine();
             
             String[] details = input.split(" ");
             
             name = details[0] + " "+ details[1];
             age = Integer.parseInt(details[2]);   
             gpa = Double.parseDouble(details[3]);  
      
             System.out.println("Student named "+ name + " is "+ age + " years old and has "+ gpa + " GPA.");
             
             scanner.close();
         
   }
}
public class RemoveCharacters
{
      public static void main(String[] args)
   {
             String input = "ABCDEFGHI";
             System.out.println("Original String: "+ input);
             System.out.println("Modified String: "+ removeThirdAndEighthChars(input));
             String shortInput = "ABCD";
             System.out.println("Original String: "+ shortInput);
             System.out.println("Modified String: "+ removeThirdAndEighthChars(shortInput));
         
   }
      public static String removeThirdAndEighthChars(String input)
   {
             if (input.length() >= 8)
      {
                    input = input.substring(0, 2) + input.substring(3, 7) + input.substring(8);
                
      }
      else if (input.length() >= 3)
      {
                    input = input.substring(0, 2) + input.substring(3);
                
      }
             return input;
         
   }
}
